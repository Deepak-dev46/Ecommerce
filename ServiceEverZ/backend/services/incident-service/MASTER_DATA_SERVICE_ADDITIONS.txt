## 2. UserController.java — paste this method before the closing brace

    /**
     * GET /api/master/users/support
     *
     * Returns all ACTIVE users with designation = SUPPORT.
     * Called by incident-service (IncidentServiceImpl) for auto-assignment.
     *
     * NEW ENDPOINT — does NOT modify any existing endpoints.
     */
    @GetMapping("/support")
    public ResponseEntity<ApiResponse<List<UserResponse>>> getSupportPersonnel() {
        log.info("GET /api/master/users/support");
        List<User> supportUsers = userRepository
                .findByDesignationAndStatus(User.Designation.SUPPORT, User.Status.ACTIVE);
        List<UserResponse> data = supportUsers.stream()
                .map(this::toResponse)
                .collect(Collectors.toList());
        ApiResponse<List<UserResponse>> response = new ApiResponse<>();
        response.setSuccess(true);
        response.setMessage("Support personnel fetched successfully");
        response.setData(data);
        response.setTimestamp(LocalDateTime.now());
        return ResponseEntity.ok(response);
    }
